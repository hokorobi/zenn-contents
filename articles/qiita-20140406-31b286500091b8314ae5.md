command! -count=1 -nargs=1 -complete=customlist,SAG_Complete SplitAndGo call SplitAndGo(<q-args>)

function! SplitAndGo(cmd)
  execute a:cmd
  if !v:count
    return
  endif
  execute "normal! " . v:count . 'G'
endfunction

function! SAG_Complete(ArgLead, CmdLine, CursorPos)
  return ['split', 'vsplit']
endfunction
